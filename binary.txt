#include <iostream>
#include <algorithm>
using namespace std;
int binary(long long A[], int low, int high, long long key);
int main()
{
int N, q;
cin>>N;
long long A[N];
for(int i=1; i<=N; i++)
cin>>A[i];
sort(A+1, A+N+1);
cin>>q;
int low, high;
low = 1;
high = N;
for(int i=1; i<=q; i++){
long long result, x;
cin>>x;
result = binary(A, low, high, x);
if(result != -1)
cout<<result<<endl;
}
return 0;
}
int binary(long long A[], int low, int high, long long key){
while(low<=high){
int mid = (low + high)/2;
if(A[mid] < key)
low = mid+1;
else if(A[mid] > key)
high = mid-1;
else
return mid;
}
return -1;
}
